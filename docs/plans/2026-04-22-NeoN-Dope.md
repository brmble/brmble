```typescript
import { useState, useEffect } from 'react';
import { INITIAL_PRODUCTION } from '../constants';

export const useGameEngine = () => {
  const [money, setMoney] = useState(250.0);
  const [production, setProduction] = useState(INITIAL_PRODUCTION);
  const [dealer, setDealer] = useState({ name: 'Thomas "G" Palmer', selling: 'weed', salesRate: 3.45 });

  useEffect(() => {
    const timer = setInterval(() => {
      setProduction(prev => {
        const next = { ...prev };
        Object.keys(next).forEach(key => { next[key].stock += next[key].rate; });
        
        const activeProd = next[dealer.selling];
        if (activeProd.stock >= dealer.salesRate) {
          activeProd.stock -= dealer.salesRate;
          setMoney(m => m + (dealer.salesRate * activeProd.price));
        }
        return next;
      });
    }, 1000);
    return () => clearInterval(timer);
  }, [dealer.selling, dealer.salesRate]);

  return { money, production, dealer, setDealer };
};
```
