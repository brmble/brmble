```typescript
import { useEffect, useRef } from 'react';
export function useInterval(callback: () => void, delay: number | null) {
  const savedCallback = useRef(callback);
  useEffect(() => { savedCallback.current = callback; }, [callback]);
  useEffect(() => {
    if (delay !== null) {
      const id = setInterval(() => savedCallback.current(), delay);
      return () => clearInterval(id);
    }
  }, [delay]);
}
```

```typescript
export const useGameEngine = () => {
  const [state, setState] = useState<GameState>(INITIAL_GAME_STATE);

  const tick = () => {
    setState(prev => {
      const next = { ...prev, production: { ...prev.production } };
      // 1. Production
      Object.keys(next.production).forEach(key => {
        next.production[key].stock += next.production[key].rate;
      });
      // 2. Atomic Sales Logic with Math.min safety
      const active = next.production[next.dealer.selling];
      const amountToSell = Math.min(active.stock, next.dealer.salesRate);
      active.stock -= amountToSell;
      next.money += (amountToSell * active.price);
      return next;
    });
  };

  const upgrade = (id: string) => {
    setState(prev => {
      const item = prev.production[id];
      if (prev.money < item.upgradeCost) return prev;
      return {
        ...prev,
        money: prev.money - item.upgradeCost,
        production: {
          ...prev.production,
          [id]: { ...item, level: item.level + 1, rate: item.rate + 0.1, upgradeCost: item.upgradeCost * 1.5 }
        }
      };
    });
  };

  useInterval(tick, 1000);
  return { state, upgrade, setDealer: (d: any) => setState(p => ({...p, dealer: d})) };
};
```
